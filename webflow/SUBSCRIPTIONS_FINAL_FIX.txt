<!-- This version manually attaches click handlers to ensure checkout works -->

<script src="https://rensto-webflow-scripts.vercel.app/shared/stripe-core.js"></script>
<script src="https://rensto-webflow-scripts.vercel.app/subscriptions/checkout.js?v=2"></script>

<script>
(function(){
  console.log('🔧 Setting up subscription buttons (FINAL FIX)...');
  
  if(!window.RenstoStripe){
    console.error('❌ RenstoStripe core not loaded');
    return;
  }

  // Wait for everything to load
  setTimeout(() => {
    const API_URL = 'https://api.rensto.com/api/stripe/checkout';
    const buttons = document.querySelectorAll('.pricing-button, .subscription-button');
    
    console.log(`Found ${buttons.length} buttons`);

    if (buttons.length === 0) {
      console.warn('⚠️ No subscription buttons found.');
      return;
    }

    // Set data attributes and attach click handlers directly
    buttons.forEach((button, index) => {
      const href = button.getAttribute('href') || button.href || '';
      
      // Extract plan from href
      let plan = null;
      const urlMatch = href.match(/[?&]plan=([^&]+)/i);
      if (urlMatch) {
        plan = urlMatch[1].toLowerCase();
      }
      
      // Fallback: infer from text or position
      if (!plan) {
        const text = (button.textContent || '').toLowerCase();
        if (text.includes('starter') || index === 0) plan = 'starter';
        else if (text.includes('pro') || text.includes('professional') || index === 1) plan = 'professional';
        else if (text.includes('enterprise') || index === 2) plan = 'enterprise';
      }
      
      // Map to tier
      const tierMap = {
        'starter': 'starter',
        'pro': 'professional',
        'professional': 'professional',
        'enterprise': 'enterprise'
      };
      
      const tier = plan ? (tierMap[plan] || 'starter') : 
                   (index === 0 ? 'starter' : index === 1 ? 'professional' : 'enterprise');
      
      const priceMap = { starter: 299, professional: 599, enterprise: 1499 };
      const price = priceMap[tier] || 299;
      
      // Set ALL required data attributes
      button.setAttribute('data-flow-type', 'subscription');
      button.setAttribute('data-tier', tier);
      button.setAttribute('data-subscription-type', 'lead-gen');
      button.setAttribute('data-page-type', 'subscriptions');
      button.setAttribute('data-price', price);
      
      console.log(`Button ${index + 1}: tier=${tier}, price=$${price}`);
      
      // Remove any existing click handlers
      const newButton = button.cloneNode(true);
      button.parentNode.replaceChild(newButton, button);
      
      // Attach click handler DIRECTLY (bypassing initCheckoutButtons)
      newButton.addEventListener('click', async function(e) {
        e.preventDefault();
        e.stopPropagation();
        
        console.log('🖱️ Button clicked:', tier, price);
        
        // Show loading state
        const originalText = newButton.textContent;
        newButton.disabled = true;
        newButton.textContent = 'Processing...';
        newButton.style.opacity = '0.6';
        
        try {
          console.log('📤 Calling API:', API_URL);
          console.log('📤 Payload:', {
            flowType: 'subscription',
            tier: tier,
            subscriptionType: 'lead-gen',
            pageType: 'subscriptions'
          });
          
          const response = await fetch(API_URL, {
            method: 'POST',
            headers: {
              'Content-Type': 'application/json'
            },
            body: JSON.stringify({
              flowType: 'subscription',
              tier: tier,
              subscriptionType: 'lead-gen',
              pageType: 'subscriptions'
            })
          });
          
          console.log('📥 Response status:', response.status);
          
          if (!response.ok) {
            const errorData = await response.json().catch(() => ({}));
            console.error('❌ API Error:', response.status, errorData);
            throw new Error(errorData.error || `API returned ${response.status}`);
          }
          
          const data = await response.json();
          console.log('✅ API Response:', data);
          
          if (data.success && data.url) {
            console.log('🚀 Redirecting to:', data.url);
            window.location.href = data.url;
          } else {
            console.error('❌ No checkout URL in response:', data);
            alert('Unable to process checkout. Please try again or contact support.');
            newButton.disabled = false;
            newButton.textContent = originalText;
            newButton.style.opacity = '1';
          }
        } catch (error) {
          console.error('❌ Checkout error:', error);
          console.error('Error details:', error.message, error.stack);
          alert('Unable to process checkout. Please try again or contact support.');
          newButton.disabled = false;
          newButton.textContent = originalText;
          newButton.style.opacity = '1';
        }
      });
    });

    console.log('✅ All buttons configured with direct click handlers');
  }, 200);
})();
</script>
