<script src="https://rensto-webflow-scripts.vercel.app/shared/stripe-core.js"></script>
<script src="https://rensto-webflow-scripts.vercel.app/subscriptions/checkout.js?v=4"></script>

<script>
(function(){
  console.log('🔧 SUBSCRIPTIONS FINAL FIX - Maximum Debug Mode...');
  
  if(!window.RenstoStripe){
    console.error('❌ CRITICAL: RenstoStripe core not loaded!');
    alert('ERROR: Stripe checkout system not loaded. Please refresh the page.');
    return;
  }

  // FORCE DEBUG MODE IMMEDIATELY
  if(window.RenstoStripe.enableDebug) {
    window.RenstoStripe.enableDebug();
    console.log('✅ Debug mode FORCED ON');
  }

  // Wait for DOM
  setTimeout(() => {
    const buttons = document.querySelectorAll('.pricing-button, .subscription-button');
    console.log(`🔍 Found ${buttons.length} buttons`);

    if (buttons.length === 0) {
      console.error('❌ NO BUTTONS FOUND!');
      return;
    }

    // Process each button
    buttons.forEach((button, index) => {
      const href = button.getAttribute('href') || button.href || '';
      
      // Extract plan
      let plan = null;
      const urlMatch = href.match(/[?&]plan=([^&]+)/i);
      if (urlMatch) plan = urlMatch[1].toLowerCase();
      
      if (!plan) {
        const text = (button.textContent || '').toLowerCase();
        if (text.includes('starter') || index === 0) plan = 'starter';
        else if (text.includes('pro') || text.includes('professional') || index === 1) plan = 'professional';
        else if (text.includes('enterprise') || index === 2) plan = 'enterprise';
      }
      
      const tierMap = { 'starter': 'starter', 'pro': 'professional', 'professional': 'professional', 'enterprise': 'enterprise' };
      const tier = plan ? (tierMap[plan] || plan) : (index === 0 ? 'starter' : index === 1 ? 'professional' : 'enterprise');
      const priceMap = { 'starter': 299, 'professional': 599, 'enterprise': 1499 };
      const price = priceMap[tier] || 299;
      
      // SET ATTRIBUTES
      button.setAttribute('data-flow-type', 'subscription');
      button.setAttribute('data-tier', tier);
      button.setAttribute('data-subscription-type', 'lead-gen');
      button.setAttribute('data-page-type', 'subscriptions');
      button.setAttribute('data-price', price.toString());

      // Verify
      const attrs = {
        'data-flow-type': button.getAttribute('data-flow-type'),
        'data-tier': button.getAttribute('data-tier'),
        'data-subscription-type': button.getAttribute('data-subscription-type'),
        'data-price': button.getAttribute('data-price')
      };
      
      console.log(`✅ Button ${index + 1} attributes:`, attrs);
      
      // REMOVE ALL EXISTING CLICK HANDLERS (to avoid conflicts)
      const newButton = button.cloneNode(true);
      button.parentNode.replaceChild(newButton, button);
      const cleanButton = newButton;
      
      // NOW ADD OUR HANDLER
      cleanButton.addEventListener('click', function(e) {
        e.preventDefault();
        e.stopPropagation();
        e.stopImmediatePropagation();
        
        console.log('\n🔵 ========================================');
        console.log(`🔵 BUTTON ${index + 1} CLICKED`);
        console.log('🔵 ========================================');
        
        // Get attributes AT CLICK TIME
        const clickAttrs = {
          'data-flow-type': this.getAttribute('data-flow-type'),
          'data-subscription-type': this.getAttribute('data-subscription-type'),
          'data-tier': this.getAttribute('data-tier'),
          'data-price': this.getAttribute('data-price')
        };
        
        console.log('📋 Attributes at click time:', clickAttrs);
        console.log('📋 Full button element:', this);
        
        // Validate
        if (!clickAttrs['data-flow-type']) {
          const msg = 'VALIDATION FAIL: Missing data-flow-type';
          console.error('❌', msg);
          alert('ERROR: ' + msg + '\n\nCheck console for details.');
          return;
        }
        
        if (!clickAttrs['data-subscription-type']) {
          const msg = 'VALIDATION FAIL: Missing data-subscription-type';
          console.error('❌', msg);
          alert('ERROR: ' + msg + '\n\nCheck console for details.');
          return;
        }
        
        if (!clickAttrs['data-tier']) {
          const msg = 'VALIDATION FAIL: Missing data-tier';
          console.error('❌', msg);
          alert('ERROR: ' + msg + '\n\nCheck console for details.');
          return;
        }
        
        console.log('✅ Pre-validation passed');
        console.log('🚀 About to call handleCheckout...');
        console.log('🔍 window.RenstoStripe:', window.RenstoStripe);
        console.log('🔍 handleCheckout function:', window.RenstoStripe.handleCheckout);
        
        // Prepare data for manual API call if needed
        const checkoutData = {
          flowType: clickAttrs['data-flow-type'],
          tier: clickAttrs['data-tier'],
          subscriptionType: clickAttrs['data-subscription-type'],
          customerEmail: '',
          metadata: {
            source: 'webflow',
            pageType: 'subscriptions',
            price: clickAttrs['data-price']
          }
        };
        
        console.log('📤 Checkout data prepared:', checkoutData);
        
        // Try calling handleCheckout
        try {
          if (window.RenstoStripe && window.RenstoStripe.handleCheckout) {
            console.log('🚀 Calling window.RenstoStripe.handleCheckout(this)...');
            
            // Wrap in try-catch to catch ANY error
            try {
              window.RenstoStripe.handleCheckout(this);
              console.log('✅ handleCheckout called (check for redirect or errors above)');
            } catch(handlerError) {
              console.error('❌ EXCEPTION in handleCheckout:', handlerError);
              console.error('Error name:', handlerError.name);
              console.error('Error message:', handlerError.message);
              console.error('Error stack:', handlerError.stack);
              
              // Try manual API call as fallback
              console.log('🔄 Attempting manual API call as fallback...');
              manualCheckout(checkoutData);
            }
          } else {
            console.error('❌ handleCheckout not available!');
            console.error('Available methods:', Object.keys(window.RenstoStripe || {}));
            alert('ERROR: Checkout handler not loaded. Please refresh the page.');
          }
        } catch(err) {
          console.error('❌ EXCEPTION in click handler:', err);
          console.error('Error name:', err.name);
          console.error('Error message:', err.message);
          console.error('Error stack:', err.stack);
          alert('ERROR: ' + err.name + ': ' + err.message + '\n\nCheck console for full details.');
        }
      }, true); // Capture phase to run first
    });

    // Manual checkout function (fallback)
    async function manualCheckout(data) {
      try {
        console.log('📡 Making manual API call...');
        console.log('📤 Request data:', data);
        console.log('📤 Request URL: https://api.rensto.com/api/stripe/checkout');
        
        const response = await fetch('https://api.rensto.com/api/stripe/checkout', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
          },
          body: JSON.stringify(data),
        });
        
        console.log('📥 Response status:', response.status);
        console.log('📥 Response headers:', [...response.headers.entries()]);
        
        const result = await response.json();
        console.log('📥 Response body:', result);
        
        if (result.success && result.url) {
          console.log('✅ SUCCESS! Redirecting to:', result.url);
          window.location.href = result.url;
        } else {
          console.error('❌ API Error:', result.error || result);
          alert('API Error: ' + (result.error || 'Unknown error') + '\n\nDetails in console.');
        }
      } catch(err) {
        console.error('❌ Network Error:', err);
        alert('Network Error: ' + err.message + '\n\nCheck console for details.');
      }
    }
    
    // Don't call initCheckoutButtons - we're handling clicks ourselves
    console.log('\n✅ SETUP COMPLETE - Click a button to test');
    console.log('📋 Handlers attached directly to buttons');
    
  }, 200);
})();
</script>
