<script>
// Mobile menu toggle functionality
(function() {
    // Wait for DOM
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', initMobileMenu);
    } else {
        initMobileMenu();
    }
    
    function initMobileMenu() {
        const toggle = document.querySelector('.mobile-menu-toggle');
        const navUl = document.querySelector('.rensto-nav ul');
        
        if (toggle && navUl) {
            toggle.addEventListener('click', function() {
                const isOpen = navUl.style.display === 'flex';
                
                if (isOpen) {
                    navUl.style.display = 'none';
                } else {
                    navUl.style.display = 'flex';
                    navUl.style.flexDirection = 'column';
                    navUl.style.position = 'absolute';
                    navUl.style.top = '70px';
                    navUl.style.left = '0';
                    navUl.style.width = '100%';
                    navUl.style.backgroundColor = 'rgba(17, 13, 40, 0.95)';
                    navUl.style.padding = '1rem 2rem';
                    navUl.style.boxShadow = '0 8px 16px rgba(0,0,0,0.2)';
                    navUl.style.zIndex = '1001';
                }
            });
        }
    }
})();
</script>
