The endpoint:

```typescript
import { NextResponse } from "next/server";
import { prisma } from "@/lib/prisma";
import { Prisma } from "@prisma/client";

interface RankingRow {
  prompt_key: string;
  version: string;
  shot_type: string;
  avg_score: number;
  sample_count: bigint;
  avg_cost: number;
}

export async function GET(request: Request) {
  try {
    const { searchParams } = new URL(request.url);
    const shotType = searchParams.get("shot_type");

    // content_entries is a Drizzle-managed table, not in Prisma schema
    // Must use $queryRaw with Prisma.sql for parameterized queries
    let rankings: RankingRow[];

    if (shotType) {
      rankings = await prisma.$queryRaw<RankingRow[]>(Prisma.sql`
        SELECT
          generation_meta->>'prompt_key' as prompt_key,
          generation_meta->>'prompt_version' as version,
          generation_meta->>'shot_type' as shot_type,
          AVG(performance_score) as avg_score,
          COUNT(*) as sample_count,
          AVG((generation_meta->>'generation_cost_usd')::float) as avg_cost
        FROM content_entries
        WHERE performance_score IS NOT NULL
          AND generation_meta IS NOT NULL
          AND generation_meta->>'shot_type' = ${shotType}
        GROUP BY 1, 2, 3
        ORDER BY avg_score DESC
      `);
    } else {
      rankings = await prisma.$queryRaw<RankingRow[]>(Prisma.sql`
        SELECT
          generation_meta->>'prompt_key' as prompt_key,
          generation_meta->>'prompt_version' as version,
          generation_meta->>'shot_type' as shot_type,
          AVG(performance_score) as avg_score,
          COUNT(*) as sample_count,
          AVG((generation_meta->>'generation_cost_usd')::float) as avg_cost
        FROM content_entries
        WHERE performance_score IS NOT NULL
          AND generation_meta IS NOT NULL
        GROUP BY 1, 2, 3
        ORDER BY avg_score DESC
      `);
    }

    // Convert BigInt sample_count to number for JSON serialization
    const serialized = rankings.map(r => ({
      prompt_key: r.prompt_key,
      version: r.version,
      shot_type: r.shot_type,
      avg_score: Number(r.avg_score),
      sample_count: Number(r.sample_count),
      avg_cost: Number(r.avg_cost),
    }));

    return NextResponse.json({ rankings: serialized });
  } catch (err: any) {
    console.error("prompt-effectiveness error:", err.message);
    return NextResponse.json({ error: "Failed to fetch prompt effectiveness data" }, { status: 500 });
  }
}
```

**Key details:**
- Uses `prisma.$queryRaw` with `Prisma.sql` tagged template for parameterized queries (prevents SQL injection)
- BigInt conversion is required because PostgreSQL COUNT(*) returns bigint, which JSON.stringify can't serialize
- Two separate queries (with/without shotType filter) avoids dynamic SQL construction — cleaner and safer
- No auth middleware shown — follow whatever pattern the existing admin routes use (check `apps/web/superseller-site/src/app/api/admin/projects/route.ts` for the auth pattern and replicate it)
- Returns empty array `{ rankings: [] }` when no data exists — this is correct behavior before any generations have run
