```typescript
export type { ProductInfo } from "../prompt-assembler";

export interface ModuleState {
  id: string;
  groupId: string;
  tenantId: string;
  moduleType: "asset-collection" | "social-setup" | "competitor-research";

export interface ModuleHandleResult {
  handled: boolean;
  response?: string;     // text to send back to the group

export interface OnboardingModule {
  moduleType: "asset-collection" | "social-setup" | "competitor-research";
  shouldActivate(products: ProductInfo[]): boolean;
  handleMessage(params: {
    groupId: string;
    tenantId: string;
    tenantSlug: string;
    messageBody: string;
    hasMedia: boolean;
    mediaUrl?: string;
    mediaType?: string;
    messageId?: string;
    senderChatId?: string;
    state: ModuleState;
  }): Promise<ModuleHandleResult>;
  getIntroMessage(tenantName: string): string;
}
```

Create `module-router.ts` with:

```typescript
export async function routeToModule(params: {
  groupId: string;
  tenantId: string;
  messageBody: string;
  hasMedia: boolean;
  mediaUrl?: string;
  mediaType?: string;
  messageId?: string;
  senderChatId?: string;
}): Promise<ModuleHandleResult>
```

Logic:
1. Call `getActiveModule(groupId)` to check if a module is already in progress
2. If active module found and phase != 'complete': import the module, call `handleMessage()`, return result
3. If no active module: call `fetchTenantProducts(tenantId)`, determine which modules are needed using each module's `shouldActivate()`, check which are already complete via `getModuleState()`, activate the next one in priority order
4. Priority order: asset-collection (quick tangible results) > social-setup > competitor-research
5. When activating: create module state with phase='intro', call `getIntroMessage()`, return as response
6. If all needed modules complete or no modules needed: return `{handled: false}` to fall through to Claude

Module activation rules (from requirements):
- asset-collection: activates for VideoForge, Winner Studio, Character-in-a-Box, Lead Pages, SocialHub (any visual product)
- social-setup: activates for SocialHub, Buzz
- competitor-research: activates for Maps/SEO, Google Maps, Lead Pages

Import modules lazily to avoid circular deps:

```typescript
const MODULE_REGISTRY = [

] as const;
```
