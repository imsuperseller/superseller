export type ShotType = 'dialogue' | 'narrative' | 'environment' | 'product' | 'social' | 'music';

    budgetTier: BudgetTier;
    prompt: string;
    imageUrl?: string;
    durationSeconds?: number;
    tenantId?: string;
    jobId?: string;
}

    premium: 999,

    provider: 'kie' | 'fal';
    modelId: string;
    useCase: string;
}>;

}
